public static void main(String[] args) {
int[] a = new int[5]; //定义数组
//输入数组的值
Scanner scanner = new Scanner(System.in);
System.out.println("请输入5个整数：");

a[i] = scanner.nextInt();
}
//冒泡排序
//1.第1趟排序
int temp; //放交换的数

for (int i = 0; i < a.length - 1 - j; i++) {
if (a[i] > a[i + 1]) {
temp = a[i];
a[i] = a[i + 1];
a[i + 1] = temp;
}
}
}
//输出数组的值

for(int i=0;i<a.length;i++){
System.out.print(a[i]+" ");
